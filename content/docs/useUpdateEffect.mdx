## Overview

A hook alike `useEffect` but skips running the effect for the first time.

[Documentation and Examples](https://ahooks.js.org/hooks/use-updateeffect)

## Examples

### Basic usage

<Preview name="useUpdateEffect"
 code={`
import useUpdateEffect from '@/src/hooks/ahooks/useUpdateEffect';

import React, { useEffect, useState } from 'react';

const Example = () => {
  const [count, setCount] = useState(0);
  const [effectCount, setEffectCount] = useState(0);
  const [updateEffectCount, setUpdateEffectCount] = useState(0);

  useEffect(() => {
    setEffectCount((c) => c + 1);
  }, [count]);

  useUpdateEffect(() => {
    setUpdateEffectCount((c) => c + 1);
    return () => {
      // do something
    };
  }, [count]); // you can include deps array if necessary

  return (
    <div>
      <p>effectCount: {effectCount}</p>
      <p>updateEffectCount: {updateEffectCount}</p>
      <p>
        <button type="button" onClick={() => setCount((c) => c + 1)}>
          reRender
        </button>
      </p>
    </div>
  );
};

export default Example;`} />

## API

The API is exactly the same as `React.useEffect`.

```typescript
useUpdateEffect(
  effect: React.EffectCallback,
  deps?: React.DependencyList,
)
```
